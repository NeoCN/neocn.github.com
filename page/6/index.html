Oracle异常

Jul 19th, 2009 | Comments

1、ORA-01791：不是 SELECTed 表达式
如果SELECT中用到DISTINCT关键字，还要使用ORDER BY 子句，那么排序的列必须是包含在SELECT列表中的字段，否则会报ORA-01791 不是SELECTed表达式。

2、ORA-00979：不是GROUP BY 表达式
聚集函数前面几个列就要在group by后面出现几列，否则出现“不是 GROUP BY 表达式”。

3、ORA-22858:：数据类型的更改无效
Oracle 数据库表字段类型直接从varchar2 修改为clob时报错：ORA-22858: 数据类型的更改无效
解决办法：可以先修改为Long类型，清空该列数据，然后再从Long修改为clob类型

4、ORA-00972：identifier is too long  标识过长
Oracle 数据库 模式对象的名字长度上限为30个字符，因此数据库表名、字段名都不能超过30个字符

IP地址与长整型互转

Jul 19th, 2009 | Comments 




import java.net.InetAddress;

import java.net.UnknownHostException;



public class IPUtil {

 /**

 * @param ip

 *            ip的字符串表示

 * @return ip地址的数值表示

 * @throws UnknownHostException

 */

 public static long ipToLong(String ip) 

     throws UnknownHostException {

     InetAddress ipAddress = InetAddress.getByName(ip);

     return ipToLong(ipAddress);

 }



 /**

 * IP地址转换到长整型

 * @param ipAddress

 *            InetAddress对象

 * @return ip地址的数值表示

 * @throws UnknownHostException

 */

 public static long ipToLong(InetAddress ipAddress) 

     throws UnknownHostException {

     byte[] ipBytes = ipAddress.getAddress();

     Long ipLong = (ipBytes[0] << 24 & 0xff000000L)

                 | (ipBytes[1] << 16 & 0xff0000L)

                 | (ipBytes[2] <<  8 & 0xff00L)

                 | (ipBytes[3] <<  0 & 0xffL);

     return ipLong;

 }



 /**

 * 由低32位二进制数构成InetAddress对象

 *

 * @param ip

 *            ip地址的数值表示

 * @return InetAddress对象

 * @throws UnknownHostException

 */

 public static InetAddress longToIP(long ip) 

     throws UnknownHostException {

     byte[] ipBytes = new byte[4];

     int i = (int) ip;// 低32位

     ipBytes[0] = (byte) ((i >> 24) & 0x000000ff);

     ipBytes[1] = (byte) ((i >> 16) & 0x000000ff);

     ipBytes[2] = (byte) ((i >> 8 ) & 0x000000ff);

     ipBytes[3] = (byte) ((i >> 0 ) & 0x000000ff);

     return InetAddress.getByAddress(ipBytes);

 }



}

function createXMLHttpRequest(){
    try{
        xmlHttp = new ActiveXObject("Microsoft.XMLHTTP");
    }catch(e){
        try{
            xmlHttp = new ActiveXObject("Msxml2.XMLHTTP");
        }catch(e){
            xmlHttp = null;
        }
    }
    if ( !xmlHttp && typeof XMLHttpRequest != "undefined" ){
        xmlHttp=new XMLHttpRequest();
    }
}
